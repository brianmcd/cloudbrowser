/**
    The cloudbrowser object that is attached to the global window object
    of every browser.
    It provides the CloudBrowser API to the instance.
    @namespace cloudbrowser
 */

(function() {
  var Authentication, BrowserAPI, CloudBrowser, Ko, ServerConfig, User, Util, cloudbrowserError;

  Ko = require('./ko');

  BrowserAPI = require('./browser');

  CloudBrowser = (function() {
    function CloudBrowser(bserver) {
      var creator;
      Object.freeze(this.__proto__);
      if (bserver.creator) {
        creator = bserver.creator;
      } else {
        creator = new User("public");
      }
      this.util = new Util(bserver.server.config.emailerConfig);
      this.currentBrowser = new BrowserAPI({
        browser: bserver,
        userCtx: creator,
        cbCtx: this
      });
      if (this.currentBrowser.getAppConfig().isAuthConfigured()) {
        this.auth = new Authentication({
          bserver: bserver,
          cbCtx: this
        });
      }
      this.serverConfig = new ServerConfig({
        cbServer: bserver.server,
        userCtx: creator,
        cbCtx: this
      });
    }


    /**
        The Application Namespace
